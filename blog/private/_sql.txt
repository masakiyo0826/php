create table users(
  id int AUTO_INCREMENT NOT NULL PRIMARY KEY,
  handle varchar(128) unique,
  user_sei varchar(128),
  user_mei varchar(128),
  password varchar(128),
  del_flg int default 0,
  upd_date datetime,
  crt_date datetime,
  del_fate datetime
);

create table blog_posts(
  id int AUTO_INCREMENT NOT NULL PRIMARY KEY,
  user_id int,
  title text,
  body text,
  del_flg int default 0,
  upd_date datetime,
  crt_date datetime,
  del_fate datetime
);
